export const client = new YourSaaSClient({

## Step 4: Create Your First User (1 minute)

Let's create a simple user registration:

```typescript
// pages/api/register.ts (or app/api/register/route.ts for App Router)
import { client } from '@/lib/yoursaas'

export async function POST(request: Request) {
  try {
    const { email, password, name } = await request.json()
    
    // Create user with YourSaaS
    const result = await client.auth.register({
      email,
      password,
      name
    })
    
    return Response.json({
      success: true,
      user: result.user,
      session: result.session
    })
  } catch (error) {
    return Response.json({
      success: false,
      error: error.message
    }, { status: 400 })
  }
}
```

## Step 5: Add Authentication (1 minute)

Create a login form component:

```tsx
// components/LoginForm.tsx
'use client'
import { useState } from 'react'

export function LoginForm() {
  const [email, setEmail] = useState('')
  const [password, setPassword] = useState('')
  const [user, setUser] = useState(null)

  const handleLogin = async (e: React.FormEvent) => {
    e.preventDefault()
    
    const response = await fetch('/api/auth/login', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ email, password })
    })
    
    const data = await response.json()
    
    if (data.success) {
      setUser(data.user)
      // Store session token
      localStorage.setItem('token', data.session.token)
    }
  }

  if (user) {
    return (
      <div className="p-6 bg-green-50 rounded-lg">
        <h2 className="text-xl font-bold text-green-800">
          Welcome back, {user.name}! ğŸ‰
        </h2>
        <p className="text-green-600">You're successfully logged in.</p>
      </div>
    )
  }

  return (
    <form onSubmit={handleLogin} className="space-y-4 max-w-md mx-auto">
      <div>
        <label className="block text-sm font-medium mb-1">Email</label>
        <input
          type="email"
          value={email}
          onChange={(e) => setEmail(e.target.value)}
          className="w-full p-2 border rounded focus:ring-2 focus:ring-blue-500"
          required
        />
      </div>
      
      <div>
        <label className="block text-sm font-medium mb-1">Password</label>
        <input
          type="password"
          value={password}
          onChange={(e) => setPassword(e.target.value)}
          className="w-full p-2 border rounded focus:ring-2 focus:ring-blue-500"
          required
        />
      </div>
      
      <button
        type="submit"
        className="w-full bg-blue-600 text-white p-2 rounded hover:bg-blue-700"
      >
        Login
      </button>
    </form>
  )
}
```

## Step 6: Make Secure API Calls (1 minute)

Add protected API functionality:

```typescript
// pages/api/profile.ts
import { client } from '@/lib/yoursaas'

export async function GET(request: Request) {
  const authHeader = request.headers.get('authorization')
  const token = authHeader?.replace('Bearer ', '')
  
  if (!token) {
    return Response.json({ error: 'Unauthorized' }, { status: 401 })
  }
  
  try {
    // Verify session and get user
    const user = await client.auth.verifySession(token)
    
    // Get additional user data
    const profile = await client.users.get(user.id)
    
    return Response.json({
      success: true,
      profile
    })
  } catch (error) {
    return Response.json({ error: 'Invalid session' }, { status: 401 })
  }
}
```

## Troubleshooting

**Not seeing your changes?** Try:

1. Restart your development server
2. Check your browser console for errors  
3. Verify your API keys are correct
4. Make sure your `.env.local` file is in the project root

**Getting API errors?** 

- Check your API key has the right permissions
- Verify your project ID is correct
- Review the [API Reference](/docs/api-reference) for detailed error codes
